RePlate Backend Implementation

Below is a complete, modular Node.js/Express backend for RePlate. It follows a clean architecture with separate configuration, models, controllers, routes, and middleware. Each file is explained briefly and shown in full.

const mongoose = require('mongoose');

const userSchema = new mongoose.Schema({
  email: { type: String, unique: true, required: true },
  password: { type: String },
  googleId: { type: String },
  roles: { type: [String], default: ['donor', 'recipient'] },
  activeRole: { type: String, enum: ['donor', 'recipient'], default: 'recipient' },
  location: {
    type: { type: String, enum: ['Point'], default: 'Point' },
    coordinates: { type: [Number] }
  },
  preferences: { type: [String], default: [] }
}, { timestamps: true });

// Geospatial index for location
userSchema.index({ location: '2dsphere' });

// Hide sensitive fields when converting to JSON
userSchema.set('toJSON', {
  transform: (doc, ret) => {
    delete ret.password;
    delete ret.googleId;
    return ret;
  }
});

module.exports = mongoose.model('User', userSchema);


Defines the User schema with fields for email, password, Google ID, roles, active role, geolocation, and preferences. It also creates a 2dsphere index on location for geo-queries. Sensitive fields (password, googleId) are removed when converting to JSON.

Middleware
middleware/authMiddleware.js
const jwt = require('jsonwebtoken');
const User = require('../models/userModel');

const authMiddleware = async (req, res, next) => {
  try {
    // Check for token in cookies or Authorization header
    const token = req.cookies.token || 
      (req.headers.authorization &&
       req.headers.authorization.startsWith('Bearer') &&
       req.headers.authorization.split(' ')[1]);
    if (!token) {
      return res.status(401).json({ message: 'Authentication token missing' });
    }
    // Verify token and attach user to request
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    const user = await User.findById(decoded.id);
    if (!user) {
      return res.status(401).json({ message: 'User not found' });
    }
    req.user = user;
    next();
  } catch (err) {
    console.error('Auth error:', err);
    res.status(401).json({ message: 'Not authorized' });
  }
};

module.exports = authMiddleware;


Verifies the JWT (from an HTTP-only cookie or Authorization header) and attaches the authenticated user to req.user.

middleware/roleMiddleware.js
const roleMiddleware = (role) => {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({ message: 'Not authorized' });
    }
    if (req.user.activeRole !== role) {
      return res.status(403).json({ message: `Requires ${role} role` });
    }
    next();
  };
};

module.exports = roleMiddleware;


Checks that the authenticated user's activeRole matches the required role for the route (e.g. "donor" or "recipient").

Controllers
controllers/authController.js
const bcrypt = require('bcrypt');
const { OAuth2Client } = require('google-auth-library');
const User = require('../models/userModel');
const { generateToken } = require('../utils/tokenUtils');

// Initialize Google OAuth2 client
const client = new OAuth2Client(process.env.GOOGLE_CLIENT_ID);

exports.signup = async (req, res) => {
  try {
    const { email, password } = req.body;
    // Basic validation
    if (!email || !password) {
      return res.status(400).json({ message: 'Email and password are required' });
    }
    // Check if user already exists
    const existingUser = await User.findOne({ email });
    if (existingUser) {
      return res.status(400).json({ message: 'User already exists' });
    }
    // Hash password and create user
    const hashedPassword = await bcrypt.hash(password, 10);
    const user = await User.create({ email, password: hashedPassword });
    // Generate JWT and set HTTP-only cookie
    const token = generateToken(user);
    res.cookie('token', token, { httpOnly: true, secure: process.env.NODE_ENV === 'production' });
    res.status(201).json({ user });
  } catch (err) {
    console.error('Signup error:', err);
    res.status(500).json({ message: 'Server error' });
  }
};

exports.login = async (req, res) => {
  try {
    const { email, password } = req.body;
    if (!email || !password) {
      return res.status(400).json({ message: 'Email and password are required' });
    }
    // Find user and verify password
    const user = await User.findOne({ email });
    if (!user || !user.password) {
      return res.status(400).json({ message: 'Invalid credentials' });
    }
    const isMatch = await bcrypt.compare(password, user.password);
    if (!isMatch) {
      return res.status(400).json({ message: 'Invalid credentials' });
    }
    // Generate JWT and set cookie
    const token = generateToken(user);
    res.cookie('token', token, { httpOnly: true, secure: process.env.NODE_ENV === 'production' });
    res.status(200).json({ user });
  } catch (err) {
    console.error('Login error:', err);
    res.status(500).json({ message: 'Server error' });
  }
};

exports.google = async (req, res) => {
  try {
    const { idToken } = req.body;
    if (!idToken) {
      return res.status(400).json({ message: 'ID token is required' });
    }
    // Verify Google ID token
    const ticket = await client.verifyIdToken({
      idToken,
      audience: process.env.GOOGLE_CLIENT_ID
    });
    const payload = ticket.getPayload();
    const { sub: googleId, email } = payload;
    if (!email) {
      return res.status(400).json({ message: 'Google account has no email' });
    }
    // Find or create user
    let user = await User.findOne({ email });
    if (user) {
      if (!user.googleId) {
        user.googleId = googleId;
        await user.save();
      }
    } else {
      user = await User.create({ email, googleId });
    }
    // Generate JWT and set cookie
    const token = generateToken(user);
    res.cookie('token', token, { httpOnly: true, secure: process.env.NODE_ENV === 'production' });
    res.status(200).json({ user });
  } catch (err) {
    console.error('Google Auth error:', err);
    res.status(500).json({ message: 'Server error' });
  }
};


Handles user signup (with bcrypt-hashed password), login, and Google OAuth login (verifying the ID token via google-auth-library). On success, it issues a JWT in an HTTP-only cookie and returns the user profile.

controllers/foodController.js
const FoodPost = require('../models/foodModel');
const Claim = require('../models/claimModel');
const matcher = require('../services/matcher');

exports.create = async (req, res) => {
  try {
    const { name, type, quantity, expiry, location } = req.body;
    // Expect location as { lat, lon }
    if (!name || !quantity || !location || typeof location.lat !== 'number' || typeof location.lon !== 'number') {
      return res.status(400).json({ message: 'Name, quantity, and valid location (lat, lon) are required' });
    }
    // Create new food post
    const foodPost = await FoodPost.create({
      donor: req.user._id,
      name,
      type,
      quantity,
      expiry,
      location: { type: 'Point', coordinates: [location.lon, location.lat] }
    });
    // AI matching to recommend recipients (top 3)
    const recommendations = await matcher.recommend(foodPost);
    res.status(201).json({ foodPost, recommendations });
  } catch (err) {
    console.error('Create FoodPost error:', err);
    res.status(500).json({ message: 'Server error' });
  }
};

exports.nearby = async (req, res) => {
  try {
    // Ensure recipient has a location
    if (!req.user.location || !req.user.location.coordinates || req.user.location.coordinates.length !== 2) {
      return res.status(400).json({ message: 'User location is required to find nearby food' });
    }
    const [lon, lat] = req.user.location.coordinates;
    const maxDistance = 10000; // 10 km
    // Geospatial query for nearby available food posts
    const nearbyFoods = await FoodPost.find({
      status: 'available',
      location: {
        $near: {
          $geometry: { type: 'Point', coordinates: [lon, lat] },
          $maxDistance: maxDistance
        }
      }
    }).limit(20);
    res.status(200).json({ foods: nearbyFoods });
  } catch (err) {
    console.error('Nearby Food error:', err);
    res.status(500).json({ message: 'Server error' });
  }
};

exports.claim = async (req, res) => {
  try {
    const foodId = req.params.id;
    const food = await FoodPost.findById(foodId);
    if (!food || food.status !== 'available') {
      return res.status(400).json({ message: 'Food not available or already claimed' });
    }
    // Claim the food
    food.status = 'claimed';
    food.claimedBy = req.user._id;
    await food.save();
    // Record the claim
    const claim = await Claim.create({ recipient: req.user._id, food: food._id });
    res.status(200).json({ message: 'Food claimed successfully', food });
  } catch (err) {
    console.error('Claim Food error:', err);
    res.status(500).json({ message: 'Server error' });
  }
};

exports.mine = async (req, res) => {
  try {
    // Get all food posts by this donor
    const myFoods = await FoodPost.find({ donor: req.user._id });
    res.status(200).json({ foods: myFoods });
  } catch (err) {
    console.error('Mine Food error:', err);
    res.status(500).json({ message: 'Server error' });
  }
};


create: Donors post food with name, type, quantity, expiry, and location (lat, lon in body). After creation, it calls the AI matcher to return top 3 recipients.

nearby: Recipients get nearby available food posts (within ~10km of their location).

claim: Recipients claim a food post (only if available), which updates the post and creates a Claim record.

mine: Donors retrieve all their own food posts.

const FoodPost = require('../models/foodModel');
const User = require('../models/userModel');

exports.overview = async (req, res) => {
  try {
    // Total meals saved (sum of claimed quantities)
    const savedResult = await FoodPost.aggregate([
      { $match: { status: 'claimed' } },
      { $group: { _id: null, total: { $sum: '$quantity' } } }
    ]);
    const totalMealsSaved = savedResult[0] ? savedResult[0].total : 0;
    // Active (available) food posts
    const activePosts = await FoodPost.countDocuments({ status: 'available' });
    // Number of users with each role
    const donorCount = await User.countDocuments({ roles: 'donor' });
    const recipientCount = await User.countDocuments({ roles: 'recipient' });
    res.status(200).json({
      totalMealsSaved,
      activePosts,
      donorCount,
      recipientCount
    });
  } catch (err) {
    console.error('Stats overview error:', err);
    res.status(500).json({ message: 'Server error' });
  }
};


Returns overall statistics (GET /stats/overview): total meals saved (sum of claimed food quantities), number of active food posts, and counts of donors and recipients.

Services
services/matcher.js
const User = require('../models/userModel');

// Helper: Calculate distance between two coordinates (Haversine formula)
function getDistanceFromLatLon(lat1, lon1, lat2, lon2) {
  function toRad(value) {
    return (value * Math.PI) / 180;
  }
  const R = 6371; // Earth radius in km
  const dLat = toRad(lat2 - lat1);
  const dLon = toRad(lon2 - lon1);
  const a =
    Math.sin(dLat/2) * Math.sin(dLat/2) +
    Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) *
    Math.sin(dLon/2) * Math.sin(dLon/2);
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
  return R * c;
}

// AI matching logic: score recipients for a given food post
const recommend = async (foodPost) => {
  // Extract food location and type
  const [foodLon, foodLat] = foodPost.location.coordinates;
  const foodType = foodPost.type;
  // Find all potential recipients
  const recipients = await User.find({ roles: 'recipient' });
  const scored = [];
  for (const rec of recipients) {
    if (!rec.location || !rec.location.coordinates || rec.location.coordinates.length < 2) {
      continue; // skip if no location
    }
    const [recLon, recLat] = rec.location.coordinates;
    // Calculate distance in km
    const distance = getDistanceFromLatLon(recLat, recLon, foodLat, foodLon);
    // Check preference match
    const preferenceMatch = rec.preferences && foodType && rec.preferences.includes(foodType);
    // Basic scoring: lower is better
    let score = distance;
    if (preferenceMatch) {
      score -= 5; // boost for preference match
    }
    scored.push({ user: rec, score, distance });
  }
  // Sort by score ascending
  scored.sort((a, b) => a.score - b.score);
  // Return top 3 recipients (with basic info)
  return scored.slice(0, 3).map(item => ({
    userId: item.user._id,
    email: item.user.email,
    distance: item.distance,
    score: item.score
  }));
};

module.exports = { recommend };


A simple matching algorithm that scores recipients based on distance and preference match, then returns the top 3. It uses the Haversine formula for distance and subtracts a bonus if the recipient’s preferences include the food type.

const jwt = require('jsonwebtoken');

const generateToken = (user) => {
  // Create JWT payload with user ID and active role
  const payload = { id: user._id, activeRole: user.activeRole };
  return jwt.sign(payload, process.env.JWT_SECRET, { expiresIn: '1d' });
};

const verifyToken = (token) => {
  return jwt.verify(token, process.env.JWT_SECRET);
};

module.exports = { generateToken, verifyToken };

This modular codebase provides a fully functional backend for RePlate, with JWT-based authentication, role-based access control, and clear separation of concerns. Each part uses async/await and contains comments for clarity. The AI matching logic is encapsulated in services/matcher.js and can be enhanced later. The structure is scalable and beginner-friendly, following best practices for a Node.js/Express/MongoDB application.
